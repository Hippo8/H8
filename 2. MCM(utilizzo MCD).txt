mcm

mcm:
        addi    sp, sp, -16
        sd      ra, 8(sp)
        sd      s1, 0(sp)

        mul     s1, a0, a1
        jal     ra, mcd         # after this, a0 has mcd
        div     a0, s1, a0

        ld      s1, 0(sp)
        ld      ra, 8(sp)
        addi    sp, sp, 16      # restore the stack pointer
        ret

.globl _start
.data
    num1: .dword 5
    num2: .dword 6
.text
_start:
    # call MCD
    la    a0, num1
    la    a1, num2
    ld    a0, 0(a0)
    ld    a1, 0(a1)     
    jal   ra, mcm

    #exit
    li    a7, 10
    ecall
