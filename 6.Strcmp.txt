strcmp:
      lbu  t1, 0(a0)                # dereference str1[i]
      lbu  t2, 0(a1)                # dereference str2[i]

      bne  t1, t2, strcmp_end       # if str2 != str1, break
      bne  t1, zero, strcmp_cont    # if str2 == str1 && str1 == \0 return 0
      add  a0, zero, zero

strcmp_cont:
      addi a0, a0, 1                # i++
      addi a1, a1, 1
      j    strcmp                  # loop

strcmp_end:
      addi a0, zero, 1              # return 1
      ret
